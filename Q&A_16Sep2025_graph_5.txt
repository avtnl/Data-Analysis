    daily_counts.groupby("volume_category")

)

# Step 2: Participation days = total - zero-participation
author_participation = {}
ordered_labels = ["M", "AB", "PB", "AvT"]

for label in ordered_labels:
    zero_days = (
        daily_counts_extended[daily_counts_extended[f"{label}_zero"]]
        .groupby("volume_category")
        .size()
        .reindex(["some", "more", "intense", "crazy"])
        .fillna(0)
    )
    participation_days = total_days - zero_days
    author_participation[label] = participation_days

# Step 3: Combine into one DataFrame with Total Days first
participation_df = pd.DataFrame(author_participation)
participation_df.insert(0, "Total Days", total_days)
participation_df = participation_df[["Total Days", "M", "AB", "PB", "AvT"]]  # Ensure column order
participation_df.index.name = "Category"

# Step 4: Plot the chart
participation_df.plot(kind="bar", figsize=(10, 6), colormap="Set2")
plt.title("Participation Days per Author by Message Volume Category")
plt.ylabel("Number of Days")
plt.xlabel("Message Volume Category")
plt.xticks(rotation=0)
plt.legend(title="Author")
plt.tight_layout()
plt.show()
